<script>
  (function () {

    var words = [
      "Backend Developer",
      "FastAPI Enthusiast",
      "PostgreSQL Developer",
      "System Design Learner",
      "Building Scalable APIs"
    ];

    var typed = document.getElementById("typed");
    var wi = 0;
    var ci = 0;
    var deleting = false;

    function typeLoop() {
      var current = words[wi];

      if (!deleting) {
        ci += 1;
        typed.textContent = current.slice(0, ci);

        if (ci === current.length) {
          deleting = true;
          setTimeout(typeLoop, 1200);
          return;
        }

      } else {

        ci -= 1;
        typed.textContent = current.slice(0, ci);

        if (ci === 0) {
          deleting = false;
          wi = (wi + 1) % words.length;
        }
      }

      setTimeout(typeLoop, deleting ? 45 : 70);
    }

    typeLoop();

  })();
</script>
